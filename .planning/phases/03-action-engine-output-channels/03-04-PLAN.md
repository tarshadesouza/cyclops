**getPrNumber** — looks up the PR number associated with a commit SHA using the official GitHub API. Returns undefined if no open PR is associated (commit on a branch not yet in a PR):

```typescript
export async function getPrNumber(
  octokit: any,
  owner: string,
  repo: string,
  sha: string
): Promise<number | undefined> {
  try {
    const resp = await (octokit as any).request(
      'GET /repos/{owner}/{repo}/commits/{commit_sha}/pulls',
      { owner, repo, commit_sha: sha, per_page: 5 }
    );
    return (resp.data as Array<{ number: number }>)[0]?.number;
  } catch {
    return undefined;
  }
}
```

```typescript
export async function handleUpsertPrComment(ctx: ActionContext): Promise<{ skipped?: boolean }> {
  const { octokit, db, finding, owner, repo } = ctx;
  const { installationId, repositoryId, sha, workflowRunId } = finding;

  // ACT-02: skip if no PR
  const prNumber = await getPrNumber(octokit, owner, repo, sha);
  if (!prNumber) {
    ctx.log.info({ sha }, 'No PR associated with commit — skipping PR comment');
    return { skipped: true };
  }

  // Load all findings for this workflow run (consolidated body)
  const allFindings = await db.finding.findMany({
    where: { installationId, workflowRunId, deletedAt: null },
  });

  const body = renderPrCommentBody(allFindings);

  const existing = await db.prComment.findUnique({
    where: { installationId_repositoryId_prNumber: { installationId, repositoryId, prNumber } },
  });

  if (existing) {
    await (octokit as any).request(
      'PATCH /repos/{owner}/{repo}/issues/comments/{comment_id}',
      { owner, repo, comment_id: existing.githubCommentId, body }
    );
    ctx.log.info({ prNumber, commentId: existing.githubCommentId }, 'PR comment updated');
  } else {
    const resp = await (octokit as any).request(
      'POST /repos/{owner}/{repo}/issues/{issue_number}/comments',
      { owner, repo, issue_number: prNumber, body }
    );
    await db.prComment.create({
      data: { installationId, repositoryId, prNumber, githubCommentId: BigInt(resp.data.id) },
    });
    ctx.log.info({ prNumber, commentId: resp.data.id }, 'PR comment created');
  }

  return {};
}
```

**Add handleUpdateCheckRun to apps/worker/src/lib/github-outputs.ts** — ACT-03, ACT-04, ACT-13.

Requirements:
- ACT-03: Create a check run (POST) with status: 'in_progress'; store cyclopsCheckRunId on Finding
- ACT-03: Complete the check run (PATCH) with conclusion: 'failure' or 'success' and markdown summary
- ACT-04: Batch annotations 50 per PATCH call (50 is GitHub's hard limit — silently truncates otherwise)
- ACT-13: Check if cyclopsCheckRunId already set on Finding before creating — use existing if present

Finding.violations is stored as JSON (Prisma Json type). Cast it as an array of objects with optional path, line, column, message fields for annotation building.

Check run name MUST be the fixed string `'Cyclops CI Analysis'` — never dynamic (Pitfall 7 from research).

```typescript
const ANNOTATION_BATCH_SIZE = 50;
const CHECK_RUN_NAME = 'Cyclops CI Analysis';

export async function handleUpdateCheckRun(ctx: ActionContext): Promise<void> {
  const { octokit, db, finding, owner, repo } = ctx;
  const { installationId, sha } = finding;

  // ACT-13: check if cyclopsCheckRunId already exists on this finding
  let checkRunId = finding.cyclopsCheckRunId ? Number(finding.cyclopsCheckRunId) : null;

  if (!checkRunId) {
    // Phase 1: create check run in-progress
    const createResp = await (octokit as any).request(
      'POST /repos/{owner}/{repo}/check-runs',
      {
        owner, repo,
        name: CHECK_RUN_NAME,
        head_sha: sha,
        status: 'in_progress',
        started_at: new Date().toISOString(),
      }
    );
    checkRunId = createResp.data.id as number;
    // Persist immediately — if worker crashes, we can update existing run on retry
    await db.finding.update({
      where: { id: finding.id },
      data: { cyclopsCheckRunId: BigInt(checkRunId) },
    });
    ctx.log.info({ checkRunId }, 'Check run created');
  }

  // Build annotations from finding.violations
  const violations = (finding.violations as Array<{ path?: string; line?: number; message?: string }>) ?? [];
  const annotations = violations
    .filter(v => v.path && v.message)
    .map(v => ({
      path: v.path!,
      start_line: v.line ?? 1,
      end_line: v.line ?? 1,
      annotation_level: 'failure' as const,
      message: v.message!,
    }));

  const summary = renderCheckRunSummary(finding);
  const conclusion = finding.confidence && finding.confidence >= 0.85 ? 'failure' : 'neutral';

  // Phase 2: complete check run with batched annotations (ACT-04)
  if (annotations.length === 0) {
    await (octokit as any).request(
      'PATCH /repos/{owner}/{repo}/check-runs/{check_run_id}',
      {
        owner, repo, check_run_id: checkRunId,
        status: 'completed',
        conclusion,
        completed_at: new Date().toISOString(),
        output: { title: 'Cyclops Analysis', summary, annotations: [] },
      }
    );
  } else {
    // Send in batches of 50 — only mark completed on the last batch
    for (let i = 0; i < annotations.length; i += ANNOTATION_BATCH_SIZE) {
      const batch = annotations.slice(i, i + ANNOTATION_BATCH_SIZE);
      const isLast = i + ANNOTATION_BATCH_SIZE >= annotations.length;
      await (octokit as any).request(
        'PATCH /repos/{owner}/{repo}/check-runs/{check_run_id}',
        {
          owner, repo, check_run_id: checkRunId,
          output: { title: 'Cyclops Analysis', summary, annotations: batch },
          ...(isLast ? {
            status: 'completed',
            conclusion,
            completed_at: new Date().toISOString(),
          } : {}),
        }
      );
    }
  }

  ctx.log.info({ checkRunId, annotationCount: annotations.length, conclusion }, 'Check run completed');
}
```
