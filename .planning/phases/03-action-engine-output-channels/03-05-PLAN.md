```typescript
// ACT-12: rate limit — max autofixRateLimit (default 3) autofix PRs per hour per repo
export async function isRateLimited(
  db: ActionContext['db'],
  installationId: number,
  repositoryId: number,
  limit: number
): Promise<boolean> {
  const oneHourAgo = new Date(Date.now() - 60 * 60 * 1000);
  const count = await db.autofixPr.count({
    where: { installationId, repositoryId, createdAt: { gte: oneHourAgo } },
  });
  return count >= limit;
}

// ACT-11, ACT-13: check if autofix PR already created for this (installationId, repositoryId, detectorType, sha)
export async function isAutofixDeduped(
  db: ActionContext['db'],
  installationId: number,
  repositoryId: number,
  detectorType: string,
  sha: string
): Promise<boolean> {
  const existing = await db.autofixPr.findUnique({
    where: { installationId_repositoryId_detectorType_sha: { installationId, repositoryId, detectorType, sha } },
  });
  return !!existing;
}
```

**Sanity check for suggestedFix content** — skip autofix if content looks like prose (not file content):

```typescript
function isValidFileContent(content: string, detectorType: 'Lint' | 'Snapshot'): boolean {
  if (!content || content.trim().length < 10) return false;
  if (detectorType === 'Snapshot') {
    // Snapshot content must start with exports[ or Object.freeze — otherwise it's prose
    return content.trimStart().startsWith('exports[') || content.trimStart().startsWith('// Jest');
  }
  // Lint: accept if content has multiple lines (real file content, not a one-liner description)
  return content.split('\n').length > 3;
}
```

**handleAutofixLint** — Git Data API 5-step chain (Pattern 4 from research):

```typescript
export async function handleAutofixLint(ctx: ActionContext): Promise<{ skipped?: boolean; reason?: string }> {
  const { octokit, db, finding, owner, repo, config } = ctx;
  const { installationId, repositoryId, detectorType, sha, ref, suggestedFix, affectedFiles } = finding;

  // Skip if no suggested fix or it fails sanity check
  if (!suggestedFix || !isValidFileContent(suggestedFix, 'Lint')) {
    ctx.log.warn({ findingId: finding.id }, 'Autofix skipped: suggestedFix empty or not valid file content');
    return { skipped: true, reason: 'no_valid_suggested_content' };
  }

  // ACT-13: dedup check
  if (await isAutofixDeduped(db, installationId, repositoryId, detectorType, sha)) {
    ctx.log.info({ sha, detectorType }, 'Autofix PR already exists for this SHA — skipping');
    return { skipped: true, reason: 'deduped' };
  }

  // ACT-12: rate limit check
  if (await isRateLimited(db, installationId, repositoryId, config.autofixRateLimit)) {
    ctx.log.warn({ repositoryId }, 'Autofix rate limit reached — skipping');
    return { skipped: true, reason: 'rate_limited' };
  }

  // Determine affected file path — use first affectedFile
  const filePath = affectedFiles[0];
  if (!filePath) {
    ctx.log.warn({ findingId: finding.id }, 'Autofix skipped: no affectedFiles');
    return { skipped: true, reason: 'no_affected_files' };
  }

  // Step 1: Get current commit to find tree SHA
  const commitResp = await (octokit as any).request(
    'GET /repos/{owner}/{repo}/git/commits/{commit_sha}',
    { owner, repo, commit_sha: sha }
  );
  const baseTreeSha: string = commitResp.data.tree.sha;

  // Step 2: Create new tree with fixed content (inline — no separate blob call needed)
  const treeResp = await (octokit as any).request(
    'POST /repos/{owner}/{repo}/git/trees',
    {
      owner, repo,
      base_tree: baseTreeSha,
      tree: [{ path: filePath, mode: '100644', type: 'blob', content: suggestedFix }],
    }
  );

  // Step 3: Create commit
  const newCommitResp = await (octokit as any).request(
    'POST /repos/{owner}/{repo}/git/commits',
    {
      owner, repo,
      message: `fix(lint): auto-fix ESLint violations [cyclops]\n\nFinding: ${finding.id}`,
      tree: treeResp.data.sha,
      parents: [sha],
    }
  );

  // Step 4: Create branch ref
  const branchName = `cyclops/autofix/lint/${sha.slice(0, 7)}-${Date.now()}`;
  await (octokit as any).request(
    'POST /repos/{owner}/{repo}/git/refs',
    { owner, repo, ref: `refs/heads/${branchName}`, sha: newCommitResp.data.sha }
  );

  // Step 5: Create PR
  const targetBranch = ref || 'main';
  const prResp = await (octokit as any).request(
    'POST /repos/{owner}/{repo}/pulls',
    {
      owner, repo,
      title: `fix(lint): auto-fix ESLint violations on ${sha.slice(0, 7)} [cyclops]`,
      body: `Automated lint fix by cyclops[bot].\n\n**Finding:** ${finding.id}\n**Confidence:** ${finding.confidence}\n**Affected file:** \`${filePath}\`\n\n> No auto-merge — please review before merging.`,
      head: branchName,
      base: targetBranch,
      draft: false,
    }
  );

  // Store in AutofixPr table (ACT-13 dedup source)
  await db.autofixPr.create({
    data: { installationId, repositoryId, detectorType, sha, branchName, prNumber: prResp.data.number },
  });

  ctx.log.info({ branchName, prNumber: prResp.data.number }, 'Autofix lint PR created');
  return {};
}
```

Snapshot autofix is conditional: only proceed if suggestedFix contains recognizable snapshot content. The Git Data API chain is identical to Lint; the difference is the commit message, branch name, PR title, and the stricter sanity check:

```typescript
export async function handleAutofixSnapshot(ctx: ActionContext): Promise<{ skipped?: boolean; reason?: string }> {
  const { octokit, db, finding, owner, repo, config } = ctx;
  const { installationId, repositoryId, detectorType, sha, ref, suggestedFix, affectedFiles } = finding;

  // Snapshot autofix has stricter feasibility: require exports[ or similar snapshot format
  if (!suggestedFix || !isValidFileContent(suggestedFix, 'Snapshot')) {
    ctx.log.warn({ findingId: finding.id }, 'Snapshot autofix skipped: no recognizable snapshot content in suggestedFix');
    return { skipped: true, reason: 'no_valid_snapshot_content' };
  }

  if (await isAutofixDeduped(db, installationId, repositoryId, detectorType, sha)) {
    ctx.log.info({ sha, detectorType }, 'Autofix snapshot PR already exists — skipping');
    return { skipped: true, reason: 'deduped' };
  }

  if (await isRateLimited(db, installationId, repositoryId, config.autofixRateLimit)) {
    ctx.log.warn({ repositoryId }, 'Autofix rate limit reached — skipping snapshot autofix');
    return { skipped: true, reason: 'rate_limited' };
  }

  const filePath = affectedFiles[0];
  if (!filePath) return { skipped: true, reason: 'no_affected_files' };

  // Same Git Data API chain as Lint (steps 1-5)
  const commitResp = await (octokit as any).request(
    'GET /repos/{owner}/{repo}/git/commits/{commit_sha}',
    { owner, repo, commit_sha: sha }
  );

  const treeResp = await (octokit as any).request(
    'POST /repos/{owner}/{repo}/git/trees',
    {
      owner, repo,
      base_tree: commitResp.data.tree.sha,
      tree: [{ path: filePath, mode: '100644', type: 'blob', content: suggestedFix }],
    }
  );

  const newCommitResp = await (octokit as any).request(
    'POST /repos/{owner}/{repo}/git/commits',
    {
      owner, repo,
      message: `test(snapshot): regenerate snapshots [cyclops]\n\nFinding: ${finding.id}`,
      tree: treeResp.data.sha,
      parents: [sha],
    }
  );

  const branchName = `cyclops/autofix/snapshot/${sha.slice(0, 7)}-${Date.now()}`;
  await (octokit as any).request(
    'POST /repos/{owner}/{repo}/git/refs',
    { owner, repo, ref: `refs/heads/${branchName}`, sha: newCommitResp.data.sha }
  );

  const targetBranch = ref || 'main';
  const prResp = await (octokit as any).request(
    'POST /repos/{owner}/{repo}/pulls',
    {
      owner, repo,
      title: `test(snapshot): regenerate snapshots on ${sha.slice(0, 7)} [cyclops]`,
      body: `Automated snapshot update by cyclops[bot].\n\n**Finding:** ${finding.id}\n**Affected file:** \`${filePath}\`\n\n> Review snapshot changes carefully before merging.`,
      head: branchName,
      base: targetBranch,
      draft: false,
    }
  );

  await db.autofixPr.create({
    data: { installationId, repositoryId, detectorType, sha, branchName, prNumber: prResp.data.number },
  });

  ctx.log.info({ branchName, prNumber: prResp.data.number }, 'Autofix snapshot PR created');
  return {};
}
```
