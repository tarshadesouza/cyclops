Create apps/worker/src/workers/action-execution.ts. This worker:
1. Validates the job payload against ActionExecutionJobSchema
2. Gets installation client (Octokit) using getInstallationClient from @ciintel/github
3. Gets tenant DB client using getTenantClient
4. Loads config via fetchConfig (kill switch source)
5. Loads the Finding from DB by findingId
6. Resolves owner/repo from the repository via GitHub API
7. Enforces kill switches (config.prComments, config.checkRuns, config.autofix, config.detectors)
8. Routes to the correct handler via HANDLERS map
9. Returns stub `{ skipped: true }` for unimplemented handlers (to be filled by later plans)

Use the Octokit type pattern from STATE.md: `type Octokit = Awaited<ReturnType<typeof getInstallationClient>>`

```typescript
import { Worker, type Job } from "bullmq";
import pino from "pino";
import { getRedis, getActionExecutionQueue, getDlqQueue } from "@ciintel/queue";
import { ActionExecutionJobSchema, type ActionExecutionJob, type ActionType } from "@ciintel/queue";
import { getInstallationClient } from "@ciintel/github";
import { getTenantClient } from "@ciintel/db";
import { fetchConfig, type CyclopsConfig } from "@ciintel/config";
import type { Finding } from "@ciintel/db";

type Octokit = Awaited<ReturnType<typeof getInstallationClient>>;

export type ActionContext = {
  job: Job<ActionExecutionJob>;
  octokit: Octokit;
  db: ReturnType<typeof getTenantClient>;
  config: CyclopsConfig;
  finding: Finding;
  owner: string;
  repo: string;
  log: pino.Logger;
};

type Handler = (ctx: ActionContext) => Promise<{ skipped?: boolean; reason?: string }>;

// HANDLERS map — stubs for Wave 3/4/5 plans to replace
const HANDLERS: Record<ActionType, Handler> = {
  'upsert-pr-comment':            async (_ctx) => ({ skipped: true, reason: 'not-yet-implemented' }),
  'update-check-run':             async (_ctx) => ({ skipped: true, reason: 'not-yet-implemented' }),
  'create-autofix-pr-lint':       async (_ctx) => ({ skipped: true, reason: 'not-yet-implemented' }),
  'create-autofix-pr-snapshot':   async (_ctx) => ({ skipped: true, reason: 'not-yet-implemented' }),
  'rerun-workflow':               async (_ctx) => ({ skipped: true, reason: 'not-yet-implemented' }),
  'cancel-workflow':              async (_ctx) => ({ skipped: true, reason: 'not-yet-implemented' }),
  'send-slack-alert':             async (_ctx) => ({ skipped: true, reason: 'not-yet-implemented' }),
  'create-github-issue':          async (_ctx) => ({ skipped: true, reason: 'not-yet-implemented' }),
};

// Kill switch map: actionType -> config flag that can suppress it
function isActionKillSwitched(actionType: ActionType, config: CyclopsConfig, detectorType: string): boolean {
  if (actionType === 'upsert-pr-comment' && !config.prComments) return true;
  if (actionType === 'update-check-run' && !config.checkRuns) return true;
  if ((actionType === 'create-autofix-pr-lint' || actionType === 'create-autofix-pr-snapshot') && !config.autofix) return true;
  // Per-detector kill switch (CFG-02)
  const detectorKey = detectorType.toLowerCase() as keyof typeof config.detectors;
  if (config.detectors[detectorKey] === false) return true;
  return false;
}

export function createActionExecutionWorker() {
  const logger = pino({ level: process.env['LOG_LEVEL'] ?? 'info' });
  const dlqQueue = getDlqQueue();

  const worker = new Worker<ActionExecutionJob>(
    'action-execution',
    async (job) => {
      const parse = ActionExecutionJobSchema.safeParse(job.data);
      if (!parse.success) {
        logger.warn({ jobId: job.id, errors: parse.error.issues }, 'Invalid action-execution job payload — skipping');
        return { skipped: true, reason: 'invalid-payload' };
      }

      const { installationId, repositoryId, findingId, actionType, sha } = parse.data;
      const jobLog = logger.child({ jobId: job.id, installationId, repositoryId, actionType, findingId });

      const octokit = await getInstallationClient(installationId);
      const db = getTenantClient(installationId);

      // Load finding from DB — all context flows from here
      const finding = await db.finding.findUnique({ where: { id: findingId } });
      if (!finding) {
        jobLog.warn('Finding not found — skipping action');
        return { skipped: true, reason: 'finding-not-found' };
      }

      // Resolve owner/repo from repository ID via GitHub API
      const repoResp = await (octokit as any).request('GET /repositories/{repository_id}', { repository_id: repositoryId });
      const owner: string = repoResp.data.owner.login;
      const repo: string = repoResp.data.name;

      // Load config — kill switches read per-job (max 60s stale), never cached at module level
      const ref = parse.data.ref ?? finding.ref ?? 'main';
      const config = await fetchConfig(octokit as any, owner, repo, ref, repositoryId);

      // Kill switch check (ACT-14)
      if (isActionKillSwitched(actionType, config, finding.detectorType)) {
        jobLog.info({ actionType, detectorType: finding.detectorType }, 'Action suppressed by kill switch');
        return { skipped: true, reason: 'kill-switch' };
      }

      const handler = HANDLERS[actionType];
      const ctx: ActionContext = { job, octokit, db, config, finding, owner, repo, log: jobLog };
      const result = await handler(ctx);

      jobLog.info({ actionType, result }, 'Action execution complete');
      return result;
    },
    {
      connection: getRedis(),
      concurrency: 5,
    }
  );

  worker.on('failed', (job, err) => {
    logger.error({ jobId: job?.id, err }, 'ActionExecutionWorker job failed');
  });

  worker.on('error', (err) => {
    logger.error({ err }, 'ActionExecutionWorker error');
  });

  // DLQ routing for exhausted retries
  worker.on('failed', async (job, err) => {
    if (!job) return;
    const maxAttempts = job.opts.attempts ?? 1;
    if (job.attemptsMade >= maxAttempts) {
      await dlqQueue.add('exhausted', {
        originalQueue: 'action-execution',
        jobId: job.id,
        jobName: job.name,
        jobData: job.data,
        error: err.message,
        failedAt: new Date().toISOString(),
      }, { removeOnComplete: false });
    }
  });

  return worker;
}
```

// Dispatcher: determine action types for this finding's detector type
function getActionTypes(detectorType: string): ActionType[] {
  const base: ActionType[] = ['update-check-run', 'upsert-pr-comment'];
  switch (detectorType) {
    case 'Lint':
      return [...base, 'create-autofix-pr-lint'];
    case 'Snapshot':
      return [...base, 'create-autofix-pr-snapshot'];
    case 'FlakyTest':
      return [...base, 'rerun-workflow'];
    case 'HangingWorkflow':
      return [...base, 'cancel-workflow'];
    case 'ExpiredSecret':
      return [...base, 'send-slack-alert', 'create-github-issue'];
    case 'Build':
    case 'TestFailure':
    case 'MissingEnv':
    default:
      return [...base, 'create-github-issue'];
  }
}
