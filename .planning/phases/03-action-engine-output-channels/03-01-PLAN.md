```prisma
model PrComment {
  id               String   @id @default(uuid())
  installationId   Int
  repositoryId     Int
  prNumber         Int
  githubCommentId  BigInt
  createdAt        DateTime @default(now())
  updatedAt        DateTime @updatedAt
  @@unique([installationId, repositoryId, prNumber], name: "installationId_repositoryId_prNumber")
  @@index([installationId])
  @@map("pr_comments")
}

model ActionDedup {
  id             String   @id @default(uuid())
  installationId Int
  repositoryId   Int
  detectorType   String
  ref            String
  actionType     String
  expiresAt      DateTime
  createdAt      DateTime @default(now())
  @@unique([installationId, repositoryId, detectorType, ref, actionType], name: "installationId_repositoryId_detectorType_ref_actionType")
  @@index([expiresAt])
  @@map("action_dedups")
}

model AutofixPr {
  id             String   @id @default(uuid())
  installationId Int
  repositoryId   Int
  detectorType   String
  sha            String
  branchName     String
  prNumber       Int
  createdAt      DateTime @default(now())
  @@unique([installationId, repositoryId, detectorType, sha], name: "installationId_repositoryId_detectorType_sha")
  @@index([installationId])
  @@map("autofix_prs")
}

model TrackedIssue {
  id                String   @id @default(uuid())
  installationId    Int
  repositoryId      Int
  detectorType      String
  ref               String
  githubIssueNumber Int
  createdAt         DateTime @default(now())
  @@unique([installationId, repositoryId, detectorType, ref], name: "installationId_repositoryId_detectorType_ref")
  @@index([installationId])
  @@map("tracked_issues")
}
```

```typescript
export const ACTION_TYPES = [
  'upsert-pr-comment',
  'update-check-run',
  'create-autofix-pr-lint',
  'create-autofix-pr-snapshot',
  'rerun-workflow',
  'cancel-workflow',
  'send-slack-alert',
  'create-github-issue',
] as const;

export type ActionType = typeof ACTION_TYPES[number];

export const ActionExecutionJobSchema = z.object({
  installationId: z.number().int().positive(),
  repositoryId:   z.number().int().positive(),
  checkRunId:     z.number().int().positive(),
  findingId:      z.string().uuid(),
  actionType:     z.enum(ACTION_TYPES),
  sha:            z.string().length(40),
  ref:            z.string().optional(),
});
export type ActionExecutionJob = z.infer<typeof ActionExecutionJobSchema>;
```

Note: `actionParams` is removed — Phase 3 handlers load all context from DB using findingId.
